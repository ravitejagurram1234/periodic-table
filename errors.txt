# Remove the SOAP 1.2 + HTTP bindings
$wsdl = [regex]::Replace($wsdl,
    '<wsdl:binding name="RequestServiceSoap12Binding".*?(?=<wsdl:binding name="RequestServiceSoap11Binding")',
    '', 'Singleline')

# Remove the HTTP + SOAP 1.2 ports
$wsdl = [regex]::Replace($wsdl,
    '<wsdl:port name="RequestServiceHttpEndpoint".*?(?=<wsdl:port name="RequestServiceHttpSoap11Endpoint")',
    '', 'Singleline')

# Save back as UTF-8 (no BOM)
$full = (Resolve-Path $path).Path
[System.IO.File]::WriteAllText($full, $wsdl, (New-Object System.Text.UTF8Encoding($false)))
